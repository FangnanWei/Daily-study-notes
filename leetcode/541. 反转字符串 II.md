class Solution {
public:
    string reverseStr(string s, int k) {
        for(int i = 0;i<s.size();i += (2*k))
            {
                int left = i;
                int right = i + k - 1;
                if((s.size()-k)<left)
                    {
                        left = i;
                        right = s.size()-1;
                    }
                while(left<right)
                    {
                        swap(s[left],s[right]);
                        left++;
                        right--;
                    }
            }
        return s;
    }
};
